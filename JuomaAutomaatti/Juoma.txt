package juomaAutomaatti; 

public class juoma {
	
	
	// Konstruktori puuttuu Eli muuttujien määrittelä tulisi olla public juoma() -nimisen rakenteen sisällä
	private int tee = 50;
	private int kahvi = 50;
	private int kaakao = 50;


	public int getTee() {
		return tee;
	}




	public void setTee(int tee) {
		this.tee = tee;
	}




	public int getKahvi() {
		return kahvi;
	}




	public void setKahvi(int kahvi) {
		this.kahvi = kahvi;
	}




	public int getKaakao() {
		return kaakao;
	}




	public void setKaakao(int kaakao) {
		this.kaakao = kaakao;
	}


	public static void main(String[] args) { // main -metodi on turha täällä
		

	}


	public void valmistaKahvi() {
		if (kahvi <= 0) {
			System.out.println("Tuoteloppu!");
			System.out.println("Valitse toinen juoma...");
			System.out.println("Tai lopeta painamalla 4 \n");
			return;
		}else
		; this.kahvi-=10; // Tässä on turha puolipiste. Se itseasiassa päättää else lauseen ja kahvia vähennetään joka tapauksessa, riippumatta if-lauseen vertailusta
	
	}
	
	public void valmistaTee() {
		if (tee <= 0) {
			System.out.println("Tuoteloppu!");
			System.out.println("Valitse toinen juoma...");
			System.out.println("Tai lopeta painamalla 4 \n");
			return;
		}else	// Tässä on turha puolipiste. Se itseasiassa päättää else lauseen ja kahvia vähennetään joka tapauksessa, riippumatta if-lauseen vertailusta
		; this.tee-=10;	
		return;
	}
	
	public void valmistaKaakao() {
		if (kaakao <= 0) {
			System.out.println("Tuoteloppu!");
			System.out.println("Valitse toinen juoma...");
			System.out.println("Tai lopeta painamalla 4\n");
			return;
		}else	
		; this.kaakao-=10; // Tässä on turha puolipiste. Se itseasiassa päättää else lauseen ja kahvia vähennetään joka tapauksessa, riippumatta if-lauseen vertailusta
	}

	public void valmistaLopeta() {
		System.out.println("Lämpimiä ja nautinnollisia juoma hetkiä :) ");
		System.exit(0);
		
	}


	public boolean onnistuuko() {
		int luku = (int)(Math.random() * 100+1) ;	
		if(luku <= 25 && luku > 0 ) {
		return false;
		}
		
		else return true;
				
	} 
	

	@Override
	public String toString() {
		return "Teetä jäljellä: " + tee + " Kahvia jäljellä: " + kahvi + " Kaakaota jäljellä " + kaakao +"\n" ;
	}


	

	

}
